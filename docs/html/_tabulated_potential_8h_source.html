345 template <bool is_periodic>
346 HOST DEVICE inline float2 linearly_interpolate(float x, float start=0.0f) const {
347 float w = (x - start) * step_inv;
348 int home = int( floorf(w) );
349 w = w - home;
350 // if (home < 0) return EnergyForce(v0[0], Vector3(0.0f));
351 if (home < 0) {
352 if (is_periodic) home += size;
353 else return make_float2(pot[0],0.0f);
354 }
355 else if (home >= size) {
356 if (is_periodic) home -= size;
357 else return make_float2(pot[size-1],0.0f);
358 }
359
360 float u0 = pot[home];
361 float du = home+1 < size ? pot[home+1]-u0 : is_periodic ? pot[0]-u0 : 0;
362
363 return make_float2(du*w+u0, du*step_inv);
364 }

416 DEVICE inline TwoVector3 get_angle_force(const Vector3& ab,
417 const Vector3& bc,
418 float energy_deriv) const {
419 // Find the distance between each pair of particles
420 float distab = ab.length2();
421 float distbc = bc.length2();
422 const float distac2 = (ab+bc).length2();
423
424 // Find the cosine of the angle we want - <ABC
425 float cos = (distab + distbc - distac2);
426
427 distab = 1.0f/sqrt(distab); //TODO: test other functions
428 distbc = 1.0f/sqrt(distbc);
429 cos *= 0.5f * distbc * distab;
430
431 // If the cosine is illegitimate, set it to 1 or -1 so that acos won't fail
432 if (cos < -1.0f) cos = -1.0f;
433 if (cos > 1.0f) cos = 1.0f;
434
435 float sin = sqrtf(1.0f - cos*cos);
436 energy_deriv /= abs(sin) > 1e-3 ? sin : 1e-3; // avoid singularity
437 if (abs(sin) <= 1e-3) {
438 printf("BAD ANGLE: sin, cos, energy_deriv, distab, distbc, distac2: (%f %f %f %f %f %f)\n",
439 sin,cos,energy_deriv,ab.length(),bc.length(),sqrt(distac2));
440 }
441
442 // Calculate the forces
443 TwoVector3 force;
444 force.v1 = (energy_deriv*distab) * (ab * (cos*distab) + bc * distbc); // force on 1st particle
445 force.v2 = -(energy_deriv*distbc) * (bc * (cos*distbc) + ab * distab); // force on last particle
446 return force;
447 }
