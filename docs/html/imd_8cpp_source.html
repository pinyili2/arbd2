40/* Only works with aligned 4-byte quantities, will cause a bus error */
41/* on some platforms if used on unaligned data. */
42void swap4_aligned(void *v, long ndata) {
43 int *data = (int *) v;
44 long i;
45 int *N;
46 for (i=0; i<ndata; i++) {
47 N = data + i;
48 *N=(((*N>>24)&0xff) | ((*N&0xff)<<24) |
49 ((*N>>8)&0xff00) | ((*N&0xff00)<<8));
50 }
51}

53static int32 imd_htonl(int32 h) {
54 int32 n;
55 ((char *)&n)[0] = (h >> 24) & 0x0FF;
56 ((char *)&n)[1] = (h >> 16) & 0x0FF;
57 ((char *)&n)[2] = (h >> 8) & 0x0FF;
58 ((char *)&n)[3] = h & 0x0FF;
59 return n;
60}

70static int32 imd_ntohl(int32 n) {
71 int32 h = 0;
72 netint net;
73
74 memcpy((void *)&net,(void *)&n, sizeof(n));
75 h |= net.highest << 24 | net.high << 16 | net.low << 8 | net.lowest;
76 return h;
77}

89static int32 imd_readn(void *s, char *ptr, int32 n) {
90 int32 nleft;
91 int32 nread;
92
93 nleft = n;
94 while (nleft > 0) {
95 if ((nread = vmdsock_read(s, ptr, nleft)) < 0) {
96 if (errno == EINTR)
97 nread = 0; /* and call read() again */
98 else
99 return -1;
100 } else if (nread == 0)
101 break; /* EOF */
102 nleft -= nread;
103 ptr += nread;
104 }
105 return n-nleft;
106}

108static int32 imd_writen(void *s, const char *ptr, int32 n) {
109 int32 nleft;
110 int32 nwritten;
111
112 nleft = n;
113 while (nleft > 0) {
114 if ((nwritten = vmdsock_write(s, ptr, nleft)) <= 0) {
115 if (errno == EINTR)
116 nwritten = 0;
117 else
118 return -1;
119 }
120 nleft -= nwritten;
121 ptr += nwritten;
122 }
123 return n;
124}

167 int rc;

169 char *buf = (char *) malloc(sizeof(char) * size);
170 fill_header((IMDheader *)buf, IMD_MDCOMM, n);

219int imd_recv_handshake(void *s) {
220 int32 buf;
221 IMDType type;
222
223 /* Wait up to 5 seconds for the handshake to come */
224 if (vmdsock_selread(s, 5) != 1) return -1;
225
226 /* Check to see that a valid handshake was received */
227 type = imd_recv_header_nolengthswap(s, &buf);
228 if (type != IMD_HANDSHAKE) return -1;
229
230 /* Check its endianness, as well as the IMD version. */
231 if (buf == IMDVERSION) {
232 if (!imd_go(s)) return 0;
233 return -1;
234 }
235
236 swap4_aligned(&buf, 1);
237 if (buf == IMDVERSION) {
238 if (!imd_go(s)) return 1;
239 }
240
241 /* We failed to determine endianness. */
242 return -1;
243}
