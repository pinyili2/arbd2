49 void makeUnitCell() {
50 float pi = 4.0f*atan(1.0f);
51
52 unitCell[0] = box.ex().length();
53 unitCell[2] = box.ey().length();
54 unitCell[5] = box.ez().length();
55
56 double bc = box.ey().dot(box.ez());
57 double ac = box.ex().dot(box.ez());
58 double ab = box.ex().dot(box.ey());
59
60 unitCell[1] = bc/unitCell[0]/unitCell[2]/pi*180.0f;
61 unitCell[3] = ac/unitCell[0]/unitCell[5]/pi*180.0f;
62 unitCell[4] = ab/unitCell[2]/unitCell[5]/pi*180.0f;
63
64 unitCell[1] = unitCell[1] > 1.0 ? 1.0 : unitCell[1] < -1.0 ? -1.0 : unitCell[1];
65 unitCell[3] = unitCell[3] > 1.0 ? 1.0 : unitCell[3] < -1.0 ? -1.0 : unitCell[3];
66 unitCell[4] = unitCell[4] > 1.0 ? 1.0 : unitCell[4] < -1.0 ? -1.0 : unitCell[4];
67 }
