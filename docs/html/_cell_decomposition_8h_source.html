168 inline int getNeighborID(const cell_t& c, int i, int j, int k) const {
169 if (i == 0 and j == 0 and k == 0)

208 // Wrap an integer with inclusive lower and upper bounds.
209 HOST DEVICE
210 static inline int wrapInt(int k, int lower, int upper) {
211 int range = upper - lower + 1;
212 if (k < lower)
213 k += range * ((lower - k) / range + 1);
214 return lower + (k - lower) % range;
215 }
216
217 // Calculate a cell's id given a position in the grid.
218 HOST DEVICE
219 static inline int getCellID(int i, int j, int k, int3 nCells) {
220 i = wrapInt(i, 0, nCells.x - 1);
221 j = wrapInt(j, 0, nCells.y - 1);
222 k = wrapInt(k, 0, nCells.z - 1);
223 return k + nCells.z * (j + (nCells.y * i));
224 }
