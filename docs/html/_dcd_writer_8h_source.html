66void pad(char *s, int len)
67{
68 int curlen;
69 int i;
70
71 curlen=strlen(s);
72
73 if (curlen>len)

75 s[len]='\0';

77 }

98 /* This function will open a dcd file for writing. It takes */
99 /* the filename to open as its only argument. It will return a */
100 /* valid file descriptor if successful or DCD_OPENFAILED if the */
101 /* open fails for some reason. If the file specifed already */
102 /* exists, it is renamed by appending .BAK to it. */
103 /* */
104 /*********************************************************************/
105 int openDcd(const char* dcdname)
106 {
107 struct stat sbuf;
108 int dcdfd;
109 char *newdcdname = 0;
110
111 if (stat(dcdname, &sbuf) == 0)
112 {
113 newdcdname = new char[strlen(dcdname)+5];
114 if(newdcdname == (char *) 0)
115 return DCD_OPENFAILED;
116 strcpy(newdcdname, dcdname);
117 strcat(newdcdname, ".BAK");
118 if(rename(dcdname, newdcdname))
119 return(DCD_OPENFAILED);
120 delete [] newdcdname;
121 }
122
123
124 if ( (dcdfd = open(dcdname, O_RDWR|O_CREAT|O_EXCL|O_LARGEFILE,
125 S_IRUSR|S_IWUSR|S_IRGRP|S_IROTH)) < 0)
126 {
127 return(DCD_OPENFAILED);
128 }
129
130 return dcdfd;
131 }

264 /* */
265 /* FUNCTION write_dcdstep */
266 /* */
267 /* INPUTS: */
268 /* fd - file descriptor for the DCD file to write to */
269 /* N - Number of atoms */
270 /* X - X coordinates */
271 /* Y - Y coordinates */
272 /* Z - Z coordinates */
273 /* unitcell - a, b, c, alpha, beta, gamma of unit cell */
274 /* */
275 /* OUTPUTS: */
276 /* none */
277 /* */
278 /* write_dcdstep writes the coordinates out for a given timestep */
279 /* to the specified DCD file. */
280 /* */
281 /************************************************************************/
282 int writeStep(int N, const float *X, const float *Y, const float *Z, const double *cell)
283
284 {
285 int NSAVC,NSTEP,NFILE;
286 int out_integer;
287
288 /* Unit cell */
289 if (cell) {
290 out_integer = 6*8;
291 NAMD_write(fd, (char *) &out_integer, sizeof(int));
292 NAMD_write(fd, (char *) cell, out_integer);
293 NAMD_write(fd, (char *) &out_integer, sizeof(int));
294 }
295
296 /* Coordinates */
297 out_integer = N*4;
298 NAMD_write(fd, (char *) &out_integer, sizeof(int));
299 NAMD_write(fd, (char *) X, out_integer);
300 NAMD_write(fd, (char *) &out_integer, sizeof(int));
301 NAMD_write(fd, (char *) &out_integer, sizeof(int));
302 NAMD_write(fd, (char *) Y, out_integer);
303 NAMD_write(fd, (char *) &out_integer, sizeof(int));
304 NAMD_write(fd, (char *) &out_integer, sizeof(int));
305 NAMD_write(fd, (char *) Z, out_integer);
306 NAMD_write(fd, (char *) &out_integer, sizeof(int));
307
308 /* don't update header until after write succeeds */
309 lseek(fd,NSAVC_POS,SEEK_SET);
310 read(fd,(void*) &NSAVC,sizeof(int));
311 lseek(fd,NSTEP_POS,SEEK_SET);
312 read(fd,(void*) &NSTEP,sizeof(int));
313 lseek(fd,NFILE_POS,SEEK_SET);
314 read(fd,(void*) &NFILE,sizeof(int));
315 NSTEP += NSAVC;
316 NFILE += 1;
317 lseek(fd,NSTEP_POS,SEEK_SET);
318 NAMD_write(fd,(char*) &NSTEP,sizeof(int));
319 lseek(fd,NFILE_POS,SEEK_SET);
320 NAMD_write(fd,(char*) &NFILE,sizeof(int));
321 lseek(fd,0,SEEK_END);
322
323 return(0);
324 }
