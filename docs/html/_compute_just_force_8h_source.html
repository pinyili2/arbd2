129 void computeFull(Vector3* force, const Vector3* pos, const int* type) const {
130 // Zero the force.
131 for (int i = 0; i < num; i++) force[i] = Vector3(0.0);
132
133 // Compute the force for all pairs.
134 for (int i = 0; i < num-1; i++) {
135 for (int j = i + 1; j < num; j++) {
136 float alpha = tableAlpha[type[i]][type[j]];
137 float eps = tableEps[type[i]][type[j]];
138 float rad6 = tableRad6[type[i]][type[j]];
139 Vector3 dr = sys->wrapDiff(pos[j] - pos[i]);
140
141 Vector3 fc = coulombForceFull(dr, alpha);
142 Vector3 fh = softcoreForce(dr, eps, rad6);
143
144 force[i] += fc + fh;
145 force[j] -= fc + fh;
146 }
147 }
148 }

151 for (int i = 0; i < num; i++) {

173 void computeTabulated(Vector3* force, const Vector3* pos, const int* type) {
174 for (int i = 0; i < num; i++) {
175 // Zero the force.
176 force[i] = Vector3(0.0);
177
178 // Loop through the neighbors.
179 for (int n = 0; n < neigh[i].length(); n++) {
180 int j = neigh[i].get(n);
181 if (j == i) continue;
182 int ind = type[i] + type[j]*numParts;
183 if (tablePot[ind] == NULL) continue;
184 Vector3 dr = sys->wrapDiff(pos[j] - pos[i]);
185
186 if (dr.length2() > cutoff2) continue;
187 Vector3 ft = tablePot[ind]->computeForce(dr);
188
189 force[i] += ft;
190 }
191 }
192 }

213 static Vector3 coulombForce(Vector3 r, float alpha, float start, float len) {
214 float d = r.length();
215
216 if (d >= start + len) return Vector3(0.0);
217 if (d <= start) {
218 Vector3 force = -alpha/(d*d*d)*r;
219 return force;
220 }
221
222 // Switching.
223 float c = alpha/(start*start);
224 Vector3 force = -c*(1.0 - (d - start)/len)/d*r;
225 return force;
226 }
227
228 static Vector3 coulombForceFull(Vector3 r, float alpha) {
229 float d = r.length();
230
231 return -alpha/(d*d*d)*r;
232 }
233
234 static Vector3 softcoreForce(Vector3 r, float eps, float rad6) {
235 const float d2 = r.length2();
236 const float d6 = d2*d2*d2;
237
238 if (d6 < rad6) return (-12*eps*(rad6*rad6/(d6*d6*d2) - rad6/(d6*d2)))*r;
239 return Vector3(0.0);
240 }
