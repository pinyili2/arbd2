178DEVICE ForceEnergy RigidBodyGrid::interpolateForceD(const Vector3 l) const {
179 Vector3 f;
180 // Vector3 l = basisInv.transform(pos - origin);
181 const int homeX = int(floor(l.x));
182 const int homeY = int(floor(l.y));
183 const int homeZ = int(floor(l.z));
184 const float wx = l.x - homeX;
185 const float wy = l.y - homeY;
186 const float wz = l.z - homeZ;
187 const float wx2 = wx*wx;
188
189 /* f.x */
190 float g3[3][4];
191 for (int iz = 0; iz < 4; iz++) {
192 float g2[2][4];
193 const int jz = (iz + homeZ - 1);
194 for (int iy = 0; iy < 4; iy++) {
195 float v[4];
196 const int jy = (iy + homeY - 1);
197 for (int ix = 0; ix < 4; ix++) {
198 const int jx = (ix + homeX - 1);
199 const int ind = jz + jy*nz + jx*nz*ny;
200 v[ix] = jz < 0 || jz >= nz || jy < 0 || jy >= ny || jx < 0 || jx >= nx ?
201 0 : val[ind];
202 }
203 const float a3 = 0.5f*(-v[0] + 3.0f*v[1] - 3.0f*v[2] + v[3])*wx2;
204 const float a2 = 0.5f*(2.0f*v[0] - 5.0f*v[1] + 4.0f*v[2] - v[3])*wx;
205 const float a1 = 0.5f*(-v[0] + v[2]);
206 g2[0][iy] = 3.0f*a3 + 2.0f*a2 + a1; /* f.x (derivative) */
207 g2[1][iy] = a3*wx + a2*wx + a1*wx + v[1]; /* f.y & f.z */
208 }
209
210 // Mix along y.
211 {
212 g3[0][iz] = 0.5f*(-g2[0][0] + 3.0f*g2[0][1] - 3.0f*g2[0][2] + g2[0][3])*wy*wy*wy +
213 0.5f*(2.0f*g2[0][0] - 5.0f*g2[0][1] + 4.0f*g2[0][2] - g2[0][3]) *wy*wy +
214 0.5f*(-g2[0][0] + g2[0][2]) *wy +
215 g2[0][1];
216 }
217
218 {
219 const float a3 = 0.5f*(-g2[1][0] + 3.0f*g2[1][1] - 3.0f*g2[1][2] + g2[1][3])*wy*wy;
220 const float a2 = 0.5f*(2.0f*g2[1][0] - 5.0f*g2[1][1] + 4.0f*g2[1][2] - g2[1][3])*wy;
221 const float a1 = 0.5f*(-g2[1][0] + g2[1][2]);
222 g3[1][iz] = 3.0f*a3 + 2.0f*a2 + a1; /* f.y */
223 g3[2][iz] = a3*wy + a2*wy + a1*wy + g2[1][1]; /* f.z */
224 }
225 }
226
227 // Mix along z.
228 f.x = -0.5f*(-g3[0][0] + 3.0f*g3[0][1] - 3.0f*g3[0][2] + g3[0][3])*wz*wz*wz +
229 -0.5f*(2.0f*g3[0][0] - 5.0f*g3[0][1] + 4.0f*g3[0][2] - g3[0][3])*wz*wz +
230 -0.5f*(-g3[0][0] + g3[0][2]) *wz -
231 g3[0][1];
232 f.y = -0.5f*(-g3[1][0] + 3.0f*g3[1][1] - 3.0f*g3[1][2] + g3[1][3])*wz*wz*wz +
233 -0.5f*(2.0f*g3[1][0] - 5.0f*g3[1][1] + 4.0f*g3[1][2] - g3[1][3])*wz*wz +
234 -0.5f*(-g3[1][0] + g3[1][2]) *wz -
235 g3[1][1];
236 f.z = -1.5f*(-g3[2][0] + 3.0f*g3[2][1] - 3.0f*g3[2][2] + g3[2][3])*wz*wz -
237 (2.0f*g3[2][0] - 5.0f*g3[2][1] + 4.0f*g3[2][2] - g3[2][3]) *wz -
238 0.5f*(-g3[2][0] + g3[2][2]);
239 float e = 0.5f*(-g3[2][0] + 3.0f*g3[2][1] - 3.0f*g3[2][2] + g3[2][3])*wz*wz*wz +
240 0.5f*(2.0f*g3[2][0] - 5.0f*g3[2][1] + 4.0f*g3[2][2] - g3[2][3]) *wz*wz +
241 0.5f*(-g3[2][0] + g3[2][2]) *wz +
242 g3[2][1];
243
244 return ForceEnergy(f,e);
245}

247DEVICE ForceEnergy RigidBodyGrid::interpolateForceDLinearly(const Vector3& l) const {
248//#ifdef cubic
249//return interpolateForceD(l);
250//#elif defined(cubic_namd)
251//return interpolateForceDnamd(l);
252//#else
253 // Find the home node.
254 const int homeX = int(floor(l.x));
255 const int homeY = int(floor(l.y));
256 const int homeZ = int(floor(l.z));
257
258 Vector3 f;
259
260 const float wx = l.x - homeX;
261 const float wy = l.y - homeY;
262 const float wz = l.z - homeZ;
263
264 float v[2][2][2];
265 for (int iz = 0; iz < 2; iz++) {
266 int jz = (iz + homeZ);
267 for (int iy = 0; iy < 2; iy++) {
268 int jy = (iy + homeY);
269 for (int ix = 0; ix < 2; ix++) {
270 int jx = (ix + homeX);
271 int ind = jz + jy*nz + jx*nz*ny;
272 v[ix][iy][iz] = jz < 0 || jz >= nz || jy < 0 || jy >= ny || jx < 0 || jx >= nx ?
273 0 : val[ind];
274 }
275 }
276 }
277
278 float g3[3][2];
279 for (int iz = 0; iz < 2; iz++) {
280 float g2[2][2];
281 for (int iy = 0; iy < 2; iy++) {
282 g2[0][iy] = (v[1][iy][iz] - v[0][iy][iz]); /* f.x */
283 g2[1][iy] = wx * (v[1][iy][iz] - v[0][iy][iz]) + v[0][iy][iz]; /* f.y & f.z */
284 }
285 // Mix along y.
286 g3[0][iz] = wy * (g2[0][1] - g2[0][0]) + g2[0][0];
287 g3[1][iz] = (g2[1][1] - g2[1][0]);
288 g3[2][iz] = wy * (g2[1][1] - g2[1][0]) + g2[1][0];
289 }
290 // Mix along z.
291 f.x = -(wz * (g3[0][1] - g3[0][0]) + g3[0][0]);
292 f.y = -(wz * (g3[1][1] - g3[1][0]) + g3[1][0]);
293 f.z = - (g3[2][1] - g3[2][0]);
294 float e = wz * (g3[2][1] - g3[2][0]) + g3[2][0];
295 return ForceEnergy(f,e);
296//#endif
297}
298DEVICE ForceEnergy RigidBodyGrid::interpolateForceDnamd(const Vector3& l) const
299{
300 Vector3 f;
301 //const Vector3 l = basisInv.transform(pos - origin);
302
303 const int homeX = int(floor(l.x));
304 const int homeY = int(floor(l.y));
305 const int homeZ = int(floor(l.z));
306 const float wx = l.x - homeX;
307 const float wy = l.y - homeY;
308 const float wz = l.z - homeZ;
309
310 Vector3 dg = Vector3(wx,wy,wz);
311
312 int inds[3];
313 inds[0] = homeX;
314 inds[1] = homeY;
315 inds[2] = homeZ;
316
317 // TODO: handle edges
318
319 // Compute b
320 float b[64]; // Matrix of values at 8 box corners
321 compute_b(b, inds);
322
323 // Compute a
324 float a[64];
325 compute_a(a, b);
326
327 // Calculate powers of x, y, z for later use
328 // e.g. x[2] = x^2
329 float x[4], y[4], z[4];
330 x[0] = 1; y[0] = 1; z[0] = 1;
331 for (int j = 1; j < 4; j++) {
332 x[j] = x[j-1] * dg.x;
333 y[j] = y[j-1] * dg.y;
334 z[j] = z[j-1] * dg.z;
335 }
336
337 float e = compute_V(a, x, y, z);
338 f = compute_dV(a, x, y, z);
339
340 //f = basisInv.transpose().transform(f);
341 return ForceEnergy(f,e);
342 }

344DEVICE float RigidBodyGrid::compute_V(float *a, float *x, float *y, float *z) const
345 {
346 float V = 0.0;
347 long int ind = 0;
348 for (int l = 0; l < 4; l++) {
349 for (int k = 0; k < 4; k++) {
350 for (int j = 0; j < 4; j++) {
351 V += a[ind] * x[j] * y[k] * z[l];
352 ind++;
353 }
354 }
355 }
356 return V;
357 }
358DEVICE Vector3 RigidBodyGrid::compute_dV(float *a, float *x, float *y, float *z) const
359 {
360 Vector3 dV = Vector3(0.0f);
361 long int ind = 0;
362 for (int l = 0; l < 4; l++) {
363 for (int k = 0; k < 4; k++) {
364 for (int j = 0; j < 4; j++) {
365 if (j > 0) dV.x += a[ind] * j * x[j-1] * y[k] * z[l]; // dV/dx
366 if (k > 0) dV.y += a[ind] * k * x[j] * y[k-1] * z[l]; // dV/dy
367 if (l > 0) dV.z += a[ind] * l * x[j] * y[k] * z[l-1]; // dV/dz
368 ind++;
369 }
370 }
371 }
372 return dV*(-1.f);
373 }
