1428 // Open the file
1429 FILE* inp = fopen(partFile.val(), "r");
1430 char line[256];
1431
1432 // If the particle file cannot be found, exit the program
1433 if (inp == NULL) {
1434 printf("ERROR: Could not open `%s'.\n", partFile.val());
1435 bool found = true;
1436 for (int i = 0; i < numParts; i++)
1437 if (part[i].num == 0)
1438 found = false;
1439 // assert(false); // TODO probably relax constraint that particle must be found; could just be in RB
1440 if (!found) {
1441 printf("ERROR: Number of particles not specified in config file.\n");
1442 exit(1);
1443 }
1444 printf("Using default coordinates file\n");
1445 return;
1446 }
1447
1448 // Our particle array has a starting capacity of 256
1449 // We will expand this later if we need to.
1450 int capacity = 256;
1451 numPartsFromFile = 0;
1452 partsFromFile = new String[capacity];
1453 indices = new int[capacity];
1454 indices[0] = 0;
1455
1456 // Get and process all lines of input
1457 while (fgets(line, 256, inp) != NULL) {
1458 // Lines in the particle file that begin with # are comments
1459 if (line[0] == '#') continue;
1460
1461 String s(line);
1462 int numTokens = s.tokenCount();
1463
1464 // Break the line down into pieces (tokens) so we can process them individually
1465 String* tokenList = new String[numTokens];
1466 s.tokenize(tokenList);
1467
1468 // Legitimate ATOM input lines have 6 tokens:
1469 // ATOM | Index | Name | X-coord | Y-coord | Z-coord
1470 // A line without exactly six tokens should be discarded.
1471 if (ParticleDynamicType == String("Langevin") || ParticleDynamicType == String("NoseHooverLangevin")) {
1472 if (numTokens != 9) {
1473 printf("Error: Invalid particle file line: %s\n", line);
1474 exit(-1);
1475 }
1476 } else {
1477 if (numTokens != 6) {
1478 printf("Error: Invalid particle file line: %s\n", line);
1479 exit(-1);
1480 }
1481 }
1482
1483 // Ensure that this particle's type was defined in the config file.
1484 // If not, discard this line.
1485 bool found;
1486 for (int j = 0; j < numParts; j++) {
1487 // If this particle type exists, add a new one to the list
1488 if (part[j].name == tokenList[2]) {
1489 found = true;
1490 part[j].num++;
1491 }
1492 }
1493
1494 // If the particle's type does not exist according to the config file, discard it.
1495 if (!found) {
1496 printf("WARNING Unknown particle type %s found and discarded.\n", tokenList[2].val());
1497 continue;
1498 }
1499
1500 // If we don't have enough room in our particle array, we need to expand it.
1501 if (numPartsFromFile >= capacity) {
1502 // Temporary pointers to the old arrays
1503 String* temp = partsFromFile;
1504 int* temp2 = indices;
1505
1506 // Double the capacity
1507 capacity *= 2;
1508
1509 // Create pointers to new arrays which are twice the size of the old ones
1510 partsFromFile = new String[capacity];
1511 indices = new int[capacity];
1512
1513 // Copy the old values into the new arrays
1514 for (int j = 0; j < numPartsFromFile; j++) {
1515 partsFromFile[j] = temp[j];
1516 indices[j] = temp2[j];
1517 }
1518
1519 // delete the old arrays
1520 delete[] temp;
1521 delete[] temp2;
1522 }
1523 // Make sure the index of this particle is unique.
1524 // NOTE: The particle list is sorted by index.
1525 bool uniqueID = true;
1526 int key = atoi(tokenList[1].val());
1527 int mid = 0;
1528
1529 // If the index is greater than the last index in the list,
1530 // this particle belongs at the end of the list. Since the
1531 // list is kept sorted, we know this is okay.
1532 if (numPartsFromFile == 0 || key > indices[numPartsFromFile - 1]) {
1533 indices[numPartsFromFile] = key;
1534 partsFromFile[numPartsFromFile++] = line;
1535 }
1536 // We need to do a binary search to figure out if
1537 // the index already exists in the list.
1538 // The assumption is that input files SHOULD have their indices sorted in
1539 // ascending order, so we shouldn't actually use the binary search
1540 // or the sort (which is pretty time consuming) very often.
1541 else {
1542 int low = 0, high = numPartsFromFile - 1;
1543
1544 while (low <= high) {
1545 mid = (int)((high - low) / 2 + low);
1546 int curr = indices[mid];
1547 if (curr < key) {
1548 low = mid + 1;
1549 } else if (curr > key) {
1550 high = mid - 1;
1551 } else {
1552 // For now, particles with non-unique IDs are simply not added to the array
1553 // Other possible approaches which are not yet implemented:
1554 // 1: Keep track of these particles and assign them new IDs after you have
1555 // already added all of the other particles.
1556 // 2: Get rid of ALL particles with that ID, even the ones that have already
1557 // been added.
1558 printf("WARNING: Non-unique ID found: %s\n", line);
1559 uniqueID = false;
1560 break;
1561 }
1562 }
1563 if (uniqueID) {
1564 // Add the particle to the end of the array, then sort it.
1565 indices[numPartsFromFile] = key;
1566 partsFromFile[numPartsFromFile++] = line;
1567 std::sort(indices, indices + numPartsFromFile);
1568 std::sort(partsFromFile, partsFromFile + numPartsFromFile, compare());
1569 }
1570 }
1571 }
