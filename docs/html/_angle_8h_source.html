37 HOST DEVICE inline float calcAngle(Vector3* pos, BaseGrid* sys) {
38 const Vector3& posa = pos[ind1];
39 const Vector3& posb = pos[ind2];
40 const Vector3& posc = pos[ind3];
41 const float distab = sys->wrapDiff(posa - posb).length();
42 const float distbc = sys->wrapDiff(posb - posc).length();
43 const float distac = sys->wrapDiff(posc - posa).length();
44 float cos = (distbc * distbc + distab * distab - distac * distac)
45 / (2.0f * distbc * distab);
46 if (cos < -1.0f) cos = -1.0f;
47 else if (cos > 1.0f) cos = 1.0f;
48 float angle = acos(cos);
49 return angle;
50 }
