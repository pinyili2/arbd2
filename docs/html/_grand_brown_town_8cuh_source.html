17inline __device__
18ForceEnergy compute_position_dependent_force(
19 const Vector3* __restrict__ pos, Vector3* __restrict__ forceInternal,
20 const int* __restrict__ type, BrownianParticleType** part,
21 const float electricField, const int scheme, const int idx)
22{
23 int t = type[idx];
24 Vector3 r0 = pos[idx];
25 const BrownianParticleType& pt = *part[t];
26 Vector3 forceExternal = Vector3(0.0f, 0.0f, pt.charge * electricField);
27
28 ForceEnergy fe(0.f, 0.f);
29 for(int i = 0; i < pt.numPartGridFiles; ++i)
30 {
31 ForceEnergy tmp(0.f, 0.f);
32 if(!scheme) {
33 BoundaryCondition bc = pt.pmf_boundary_conditions[i];
34 INTERPOLATE_FORCE(tmp, pt.pmf[i]->interpolateForceDLinearly, bc, r0)
35 } else
36 tmp = pt.pmf[i]->interpolateForceD(r0);
37 fe.f += tmp.f * pt.pmf_scale[i];
38 fe.e += tmp.e * pt.pmf_scale[i];
39 }
40 // if(get_energy)
41 // energy[idx] += fe.e;
42
43#ifndef FORCEGRIDOFF
44 // Add a force defined via 3D FORCE maps (not 3D potential maps)
45 if(!scheme)
46 {
47 if (pt.forceXGrid != NULL) fe.f.x += pt.forceXGrid->interpolatePotentialLinearly(r0);
48 if (pt.forceYGrid != NULL) fe.f.y += pt.forceYGrid->interpolatePotentialLinearly(r0);
49 if (pt.forceZGrid != NULL) fe.f.z += pt.forceZGrid->interpolatePotentialLinearly(r0);
50 }
51 else
52 {
53 if (pt.forceXGrid != NULL) fe.f.x += pt.forceXGrid->interpolatePotential(r0);
54 if (pt.forceYGrid != NULL) fe.f.y += pt.forceYGrid->interpolatePotential(r0);
55 if (pt.forceZGrid != NULL) fe.f.z += pt.forceZGrid->interpolatePotential(r0);
56 }
57#endif
58 fe.f = fe.f + forceExternal;
59 return fe;
60}

63////The kernel is for Nose-Hoover Langevin dynamics
64template <bool need_position_dep_force = false>
65__global__ void
66updateKernelNoseHooverLangevin(Vector3* __restrict__ pos, Vector3* __restrict__ momentum, float* random,
67 Vector3* __restrict__ forceInternal, int type[], BrownianParticleType* part[],
68 float kT, BaseGrid* kTGrid, float electricField, int tGridLength, float timestep,
69 int num, int num_rb_attached_particles, BaseGrid* sys, Random* randoGen, int numReplicas, int scheme)
70{
71 int idx = blockIdx.x * blockDim.x + threadIdx.x;
72 if (idx < num * numReplicas)
73 {
74 idx = (idx % num) + (idx/num) * (num+num_rb_attached_particles);
75
76 ForceEnergy fe;
77 if (need_position_dep_force) {
78 fe = compute_position_dependent_force(
79 pos, forceInternal, type, part, electricField, scheme, idx );
80 }
81
82 int t = type[idx];
83 Vector3 r0 = pos[idx];
84 Vector3 p0 = momentum[idx];
85 float ran = random[idx];
86
87 const BrownianParticleType& pt = *part[t];
88
89 if (need_position_dep_force) {
90 forceInternal[idx] = forceInternal[idx] + fe.f;
91 }
92 Vector3 force = forceInternal[idx];
93 #ifdef Debug
94 forceInternal[idx] = -force;
95 #endif
96 // Get local kT value
97 float kTlocal;
98 if(!scheme)
99 kTlocal = (tGridLength == 0) ? kT : kTGrid->interpolatePotentialLinearly(r0); /* periodic */
100 else
101 kTlocal = (tGridLength == 0) ? kT : kTGrid->interpolatePotential(r0); /* periodic */
102
103 // Update the particle's position using the calculated values for time, force, etc.
104 float mass = pt.mass;
105 float mu = pt.mu;
106 Vector3 gamma = pt.transDamping;
107 float rando = (*randoGen).gaussian(idx, num * numReplicas);
108
109 float tmp = sqrtf(kTlocal * (1.f - expf(-2.f * gamma.x * timestep)) / mu);
110
111 if (pt.diffusionGrid != NULL)
112 {
113
114 Vector3 gridCenter = pt.diffusionGrid->origin +
115 pt.diffusionGrid->basis.transform( Vector3(0.5*pt.diffusionGrid->nx, 0.5*pt.diffusionGrid->ny,
116 0.5*pt.diffusionGrid->nz));
117 Vector3 p2 = r0 - gridCenter;
118 p2 = sys->wrapDiff( p2 ) + gridCenter;
119 ForceEnergy diff;
120 if(!scheme)
121 diff = pt.diffusionGrid->interpolateForceDLinearly<periodic>(p2);
122 else
123 diff = pt.diffusionGrid->interpolateForceD(p2);
124 gamma = Vector3(kTlocal / (mass * diff.e));
125 }
126
127 #ifdef MDSTEP
128 force = Vector3(-r0.x, -r0.y, -r0.z);
129 #endif
130
131 p0 = p0 + 0.5f * timestep * force * Unit1;
132
133 r0 = r0 + 0.5f * timestep * p0 * 1e4 / mass;
134 //r0 = sys->wrap(r0);
135
136 ran = ran + 0.5f * (p0.length2() / mass * 0.238845899f - 3.f * kTlocal) / mu;
137
138 p0 = expf(-0.5f * ran) * p0;
139
140 ran = expf(-gamma.x * timestep) * ran + tmp * rando;
141
142 p0 = expf(-0.5f * ran) * p0;
143
144 ran = ran + 0.5f * (p0.length2() / mass * 0.238845899 - 3.f * kTlocal) / mu;
145
146 r0 = r0 + 0.5f * timestep * p0 * 1e4 / mass;
147 r0 = sys->wrap(r0);
148
149 pos[idx] = r0;
150 momentum[idx] = p0;
151 random[idx]= ran;
152 }
153}

158//please infer http://www.MolecularDynamics.info
159//The original BBK kernel is no longer used since the random numbers should be reused

162template <bool need_position_dep_force = false>
163__global__ void updateKernelBAOAB(Vector3* pos, Vector3* momentum, Vector3* __restrict__ forceInternal,
164 int type[], BrownianParticleType* part[], float kT, BaseGrid* kTGrid,
165 float electricField,int tGridLength, float timestep,
166 int num, int num_rb_attached_particles, BaseGrid* sys,
167 Random* randoGen, int numReplicas, int scheme)
168{
169 int idx = blockIdx.x * blockDim.x + threadIdx.x;
170
171
172 if (idx < num * numReplicas)
173 {
174 idx = (idx % num) + (idx/num) * (num+num_rb_attached_particles);
175
176 ForceEnergy fe;
177 if (need_position_dep_force) {
178 fe = compute_position_dependent_force(
179 pos, forceInternal, type, part, electricField, scheme, idx );
180 }
181 // if (get_energy) energy[idx] += fe.e;
182
183 int t = type[idx];
184 Vector3 r0 = pos[idx];
185 Vector3 p0 = momentum[idx];
186 const BrownianParticleType& pt = *part[t];
187
188 if (need_position_dep_force) { // only needed at ts == 1
189 forceInternal[idx] = forceInternal[idx] + fe.f;
190 }
191 Vector3 force = forceInternal[idx];
192
193#ifdef Debug
194 forceInternal[idx] = -force;
195#endif
196
197 // Get local kT value
198 float kTlocal;
199 if(!scheme)
200 kTlocal = (tGridLength == 0) ? kT : kTGrid->interpolatePotentialLinearly(r0); /* periodic */
201 else
202 kTlocal = (tGridLength == 0) ? kT : kTGrid->interpolatePotential(r0); /* periodic */
203
204 // Update the particle's position using the calculated values for time, force, etc.
205 float mass = pt.mass;
206 Vector3 gamma = pt.transDamping;
207 Vector3 rando = (*randoGen).gaussian_vector(idx, num * numReplicas);
208 //printf("%f %f %f\n", rando.x, rando.y, rando.z);
209 if (pt.diffusionGrid != NULL)
210 {
211
212 Vector3 gridCenter = pt.diffusionGrid->origin +
213 pt.diffusionGrid->basis.transform( Vector3(0.5*pt.diffusionGrid->nx, 0.5*pt.diffusionGrid->ny,
214 0.5*pt.diffusionGrid->nz));
215 Vector3 p2 = r0 - gridCenter;
216 p2 = sys->wrapDiff( p2 ) + gridCenter;
217 ForceEnergy diff;
218 if(!scheme)
219 diff = pt.diffusionGrid->interpolateForceDLinearly<periodic>(p2);
220 else
221 diff = pt.diffusionGrid->interpolateForceD(p2);
222 gamma = Vector3(kTlocal / (mass * diff.e));
223 }
224
225 #ifdef MDSTEP
226 force = Vector3(-r0.x, -r0.y, -r0.z);
227 #endif
228
229 p0 = p0 + 0.5f * timestep * force * Unit1;
230 r0 = r0 + 0.5f * timestep / mass * p0 * 1e4;
231
232 p0.x = expf(-timestep * gamma.x) * p0.x + sqrtf(mass * kTlocal * (1.f-expf(-2.f*timestep*gamma.x))) * rando.x * Unit2;
233 p0.y = expf(-timestep * gamma.y) * p0.y + sqrtf(mass * kTlocal * (1.f-expf(-2.f*timestep*gamma.y))) * rando.y * Unit2;
234 p0.z = expf(-timestep * gamma.z) * p0.z + sqrtf(mass * kTlocal * (1.f-expf(-2.f*timestep*gamma.z))) * rando.z * Unit2;
235
236 r0 = r0 + 0.5f * timestep * p0 * 1e4 / mass;
237 r0 = sys->wrap(r0);
238
239 pos[idx] = r0;
240 momentum[idx] = p0;
241
242 //if(idx == 0)
243 // printf("%f %f %f\n", pos[idx].x,pos[idx].y,pos[idx].z);
244 }

410__device__
411inline Vector3 step(Vector3& r0, float kTlocal, Vector3 force, float diffusion,
412 Vector3 diffGrad, float timestep, BaseGrid *sys,
413 Random *randoGen, int num) {
414 const int idx = static_cast<int>(blockIdx.x * blockDim.x + threadIdx.x);
415 // TODO: improve performance by storing state locally, then sending it back to GPU
416 Vector3 rando = (*randoGen).gaussian_vector(idx, num);
417
418 diffusion *= timestep;
419 Vector3 r = r0 + (diffusion / kTlocal) * force
420 + timestep * diffGrad
421 + sqrtf(2.0f * diffusion) * rando;
422 // Wrap about periodic boundaries
423 return sys->wrap(r);
