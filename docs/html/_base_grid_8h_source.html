350 HOST DEVICE inline float interpolatePotential(const Vector3& pos) const {
351 // Find the home node.
352 Vector3 l = basisInv.transform(pos - origin);
353
354 const int homeX = int(floor(l.x));
355 const int homeY = int(floor(l.y));
356 const int homeZ = int(floor(l.z));
357 const float wx = l.x - homeX;
358 const float wy = l.y - homeY;
359 const float wz = l.z - homeZ;
360 const float wx2 = wx*wx;
361 const float wy2 = wy*wy;
362 const float wz2 = wz*wz;
363
364 float g3[4];
365 for (int iz = 0; iz < 4; iz++) {
366 float g2[4];
367 const int jz = (iz + homeZ - 1);
368 for (int iy = 0; iy < 4; iy++) {
369 float v[4];
370 const int jy = (iy + homeY - 1);
371 for (int ix = 0; ix < 4; ix++) {
372 const int jx = (ix + homeX - 1);
373 const int ind = jz + jy*nz + jx*nz*ny;
374 v[ix] = jz < 0 || jz >= nz || jy < 0 || jy >= ny || jx < 0 || jx >= nx ?
375 0 : val[ind];
376 }
377 g2[iy] = 0.5f*(-v[0] + 3.0f*v[1] - 3.0f*v[2] + v[3])*wx2*wx +
378 0.5f*(2.0f*v[0] - 5.0f*v[1] + 4.0f*v[2] - v[3]) *wx2 +
379 0.5f*(-v[0] + v[2]) *wx +
380 v[1];
381 }
382
383 // Mix along y.
384 g3[iz] = 0.5f*(-g2[0] + 3.0f*g2[1] - 3.0f*g2[2] + g2[3])*wy2*wy +
385 0.5f*(2.0f*g2[0] - 5.0f*g2[1] + 4.0f*g2[2] - g2[3]) *wy2 +
386 0.5f*(-g2[0] + g2[2]) *wy +
387 g2[1];
388 }
389 // Mix along z.
390 const float e = 0.5f*(-g3[0] + 3.0f*g3[1] - 3.0f*g3[2] + g3[3])*wz2*wz +
391 0.5f*(2.0f*g3[0] - 5.0f*g3[1] + 4.0f*g3[2] - g3[3]) *wz2 +
392 0.5f*(-g3[0] + g3[2]) *wz +
393 g3[1];
394 return e;
395 }

416

418 const float wy = l.y - homeY;
419 const float wz = l.z - homeZ;
420

422 for (int iz = 0; iz < 2; iz++) {
423 int jz = (iz + homeZ);
424 for (int iy = 0; iy < 2; iy++) {

437 float g2[2];

451 DEVICE inline ForceEnergy interpolateForceD(const Vector3& pos) const {
452 Vector3 f;
453 const Vector3 l = basisInv.transform(pos - origin);
454
455 const int homeX = int(floor(l.x));
456 const int homeY = int(floor(l.y));
457 const int homeZ = int(floor(l.z));
458 const float wx = l.x - homeX;
459 const float wy = l.y - homeY;
460 const float wz = l.z - homeZ;
461 const float wx2 = wx*wx;
462
463 /* f.x */
464 float g3[3][4];
465 for (int iz = 0; iz < 4; iz++) {
466 float g2[2][4];
467 const int jz = (iz + homeZ - 1);
468 for (int iy = 0; iy < 4; iy++) {
469 float v[4];
470 const int jy = (iy + homeY - 1);
471 for (int ix = 0; ix < 4; ix++) {
472 const int jx = (ix + homeX - 1);
473 const int ind = jz + jy*nz + jx*nz*ny;
474 v[ix] = jz < 0 || jz >= nz || jy < 0 || jy >= ny || jx < 0 || jx >= nx ?
475 0 : val[ind];
476 }
477 const float a3 = 0.5f*(-v[0] + 3.0f*v[1] - 3.0f*v[2] + v[3])*wx2;
478 const float a2 = 0.5f*(2.0f*v[0] - 5.0f*v[1] + 4.0f*v[2] - v[3])*wx;
479 const float a1 = 0.5f*(-v[0] + v[2]);
480 g2[0][iy] = 3.0f*a3 + 2.0f*a2 + a1; /* f.x (derivative) */
481 g2[1][iy] = a3*wx + a2*wx + a1*wx + v[1]; /* f.y & f.z */
482 }
483
484 // Mix along y.
485 {
486 g3[0][iz] = 0.5f*(-g2[0][0] + 3.0f*g2[0][1] - 3.0f*g2[0][2] + g2[0][3])*wy*wy*wy +
487 0.5f*(2.0f*g2[0][0] - 5.0f*g2[0][1] + 4.0f*g2[0][2] - g2[0][3]) *wy*wy +
488 0.5f*(-g2[0][0] + g2[0][2]) *wy +
489 g2[0][1];
490 }
491
492 {
493 const float a3 = 0.5f*(-g2[1][0] + 3.0f*g2[1][1] - 3.0f*g2[1][2] + g2[1][3])*wy*wy;
494 const float a2 = 0.5f*(2.0f*g2[1][0] - 5.0f*g2[1][1] + 4.0f*g2[1][2] - g2[1][3])*wy;
495 const float a1 = 0.5f*(-g2[1][0] + g2[1][2]);
496 g3[1][iz] = 3.0f*a3 + 2.0f*a2 + a1; /* f.y */
497 g3[2][iz] = a3*wy + a2*wy + a1*wy + g2[1][1]; /* f.z */
498 }
499 }
500
501 // Mix along z.
502 f.x = -0.5f*(-g3[0][0] + 3.0f*g3[0][1] - 3.0f*g3[0][2] + g3[0][3])*wz*wz*wz +
503 -0.5f*(2.0f*g3[0][0] - 5.0f*g3[0][1] + 4.0f*g3[0][2] - g3[0][3])*wz*wz +
504 -0.5f*(-g3[0][0] + g3[0][2]) *wz -
505 g3[0][1];
506 f.y = -0.5f*(-g3[1][0] + 3.0f*g3[1][1] - 3.0f*g3[1][2] + g3[1][3])*wz*wz*wz +
507 -0.5f*(2.0f*g3[1][0] - 5.0f*g3[1][1] + 4.0f*g3[1][2] - g3[1][3])*wz*wz +
508 -0.5f*(-g3[1][0] + g3[1][2]) *wz -
509 g3[1][1];
510 f.z = -1.5f*(-g3[2][0] + 3.0f*g3[2][1] - 3.0f*g3[2][2] + g3[2][3])*wz*wz -
511 (2.0f*g3[2][0] - 5.0f*g3[2][1] + 4.0f*g3[2][2] - g3[2][3]) *wz -
512 0.5f*(-g3[2][0] + g3[2][2]);
513 float e = 0.5f*(-g3[2][0] + 3.0f*g3[2][1] - 3.0f*g3[2][2] + g3[2][3])*wz*wz*wz +
514 0.5f*(2.0f*g3[2][0] - 5.0f*g3[2][1] + 4.0f*g3[2][2] - g3[2][3]) *wz*wz +
515 0.5f*(-g3[2][0] + g3[2][2]) *wz +
516 g3[2][1];
517
518 f = basisInv.transpose().transform(f);
519 return ForceEnergy(f,e);
520
521 }

537 if (homeX < -1 || homeX >= nx+1 ||
538 homeY < -1 || homeY >= ny+1 ||
539 homeZ < -1 || homeZ >= nz+1)
540 return ForceEnergy();

542
543 float v[2][2][2];
544 for (int iz = 0; iz < 2; iz++) {
545 int jz = (iz + homeZ);
546 for (int iy = 0; iy < 2; iy++) {
547 int jy = (iy + homeY);
548 for (int ix = 0; ix < 2; ix++) {
549 int jx = (ix + homeX);
550 int ind;
551 switch (bc) {
552 case dirichlet:
553 ind = jz + jy*nz + jx*nz*ny;
554 v[ix][iy][iz] =
555 jz < 0 || jz >= nz ||
556 jy < 0 || jy >= ny ||
557 jx < 0 || jx >= nx ?
558 0 : val[ind];
559 break;
560 case neumann:
561 ind =
562 (jz < 0 ? 0 : jz >= nz ? nz-1 : jz) +
563 (jy < 0 ? 0 : jy >= ny ? ny-1 : jy)*nz +
564 (jx < 0 ? 0 : jx >= nx ? nx-1 : jx)*nz*ny;
565 v[ix][iy][iz] = val[ind];
566 break;
567 case periodic:
568 ind =
569 (jz < 0 ? nz-1 : jz >= nz ? 0 : jz) +
570 (jy < 0 ? ny-1 : jy >= ny ? 0 : jy)*nz +
571 (jx < 0 ? nx-1 : jx >= nx ? 0 : jx)*nz*ny;
572 v[ix][iy][iz] = val[ind];
573 break;
574 }
575 }
576 }
577 }
578
579 float g3[3][2];
580 for (int iz = 0; iz < 2; iz++) {
581 float g2[2][2];
582 for (int iy = 0; iy < 2; iy++) {
583 g2[0][iy] = (v[1][iy][iz] - v[0][iy][iz]); /* f.x */
584 g2[1][iy] = wx * (v[1][iy][iz] - v[0][iy][iz]) + v[0][iy][iz]; /* f.y & f.z */
585 }
586 // Mix along y.
587 g3[0][iz] = wy * (g2[0][1] - g2[0][0]) + g2[0][0];
588 g3[1][iz] = (g2[1][1] - g2[1][0]);
589 g3[2][iz] = wy * (g2[1][1] - g2[1][0]) + g2[1][0];
590 }
591 // Mix along z.
592 Vector3 f;
593 f.x = -(wz * (g3[0][1] - g3[0][0]) + g3[0][0]);
594 f.y = -(wz * (g3[1][1] - g3[1][0]) + g3[1][0]);
595 f.z = - (g3[2][1] - g3[2][0]);
596
597 f = basisInv.transpose().transform(f);
598 float e = wz * (g3[2][1] - g3[2][0]) + g3[2][0];
599 return ForceEnergy(f,e);
600 }
601 DEVICE inline ForceEnergy interpolateForceDnamd(const Vector3& pos) const {

626 float a[64];

644 DEVICE inline float compute_V(float *a, float *x, float *y, float *z) const
645 {
646 float V = 0.0;

648 for (int l = 0; l < 4; l++) {
649 for (int k = 0; k < 4; k++) {
650 for (int j = 0; j < 4; j++) {
651 V += a[ind] * x[j] * y[k] * z[l];
652 ind++;
653 }
654 }
655 }
656 return V;
657 }
658 DEVICE inline Vector3 compute_dV(float *a, float *x, float *y, float *z) const

660 Vector3 dV = Vector3(0.0f);

662 for (int l = 0; l < 4; l++) {
663 for (int k = 0; k < 4; k++) {
664 for (int j = 0; j < 4; j++) {
665 if (j > 0) dV.x += a[ind] * j * x[j-1] * y[k] * z[l]; // dV/dx

667 if (l > 0) dV.z += a[ind] * l * x[j] * y[k] * z[l-1]; // dV/dz
668 ind++;
669 }
670 }
671 }
672 return dV*(-1.f);

918 inline virtual Vector3 interpolateForce(Vector3 pos) const {
919 Vector3 f;
920 Vector3 l = basisInv.transform(pos - origin);
921 int homeX = int(floor(l.x));
922 int homeY = int(floor(l.y));
923 int homeZ = int(floor(l.z));
924 // Get the array jumps with shifted indices.
925 int jump[3];
926 jump[0] = nz*ny;
927 jump[1] = nz;
928 jump[2] = 1;
929 // Shift the indices in the home array.
930 int home[3];
931 home[0] = homeX;
932 home[1] = homeY;
933 home[2] = homeZ;
934
935 // Shift the indices in the grid dimensions.
936 int g[3];
937 g[0] = nx;
938 g[1] = ny;
939 g[2] = nz;
940
941 // Get the interpolation coordinates.
942 float w[3];
943 w[0] = l.x - homeX;
944 w[1] = l.y - homeY;
945 w[2] = l.z - homeZ;
946 // Find the values at the neighbors.
947 float g1[4][4][4];
948 //RBTODO parallelize?
949 for (int ix = 0; ix < 4; ix++) {
950 for (int iy = 0; iy < 4; iy++) {
951 for (int iz = 0; iz < 4; iz++) {
952 // Wrap around the periodic boundaries.
953 int jx = ix-1 + home[0];
954 jx = wrap(jx, g[0]);
955 int jy = iy-1 + home[1];
956 jy = wrap(jy, g[1]);
957 int jz = iz-1 + home[2];
958 jz = wrap(jz, g[2]);
959 int ind = jz*jump[2] + jy*jump[1] + jx*jump[0];
960 g1[ix][iy][iz] = val[ind];
961 }
962 }
963 }
964 f.x = interpolateDiffX(pos, w, g1);
965 f.y = interpolateDiffY(pos, w, g1);
966 f.z = interpolateDiffZ(pos, w, g1);
967 Vector3 f1 = basisInv.transpose().transform(f);
968 return f1;
969 }
