76int Scatter::countCoordinates(const char* fileName) {
77 int nRead;
78 int n = 0;
79 float x, y, z;
80 char line[256];
81
82 // Open the file.
83 FILE* inp = fopen(fileName,"r");
84 if (inp == NULL) {
85 printf("Scatter:countCoordinates Couldn't open file %s\n.",fileName);
86 exit(-1);
87 }
88
89 while (fgets(line, 256, inp) != NULL) {
90 // Ignore comments.
91 int len = strlen(line);
92 if (line[0] == '#') continue;
93 if (len < 2) continue;
94
95 // Read values.
96 nRead = sscanf(line, "%f %f %f", &x, &y, &z);
97 if (nRead >= 3) n++;
98 }
99
100 fclose(inp);

102}
