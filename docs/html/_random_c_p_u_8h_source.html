92 // advance generator by one (seed = seed * mult + add, to 48 bits)
93 HOST DEVICE inline void skip(void) {
94 rand48_seed = ( rand48_seed * rand48_mult + rand48_add ) & RAND48_MASK;

97 // split into numStreams different steams and take stream iStream
98 void split(int iStream, int numStreams) {
99 int i;
100
101 // make sure that numStreams is odd to ensure maximum period
102 numStreams |= 1;
103
104 // iterate to get to the correct stream
105 for ( i = 0; i < iStream; ++i ) skip();
106
107 // save seed and add so we can use skip() for our calculations
108 int64 save_seed = rand48_seed;
109
110 // calculate c *= ( 1 + a + ... + a^(numStreams-1) )
111 rand48_seed = rand48_add;
112 for ( i = 1; i < numStreams; ++i ) skip();
113 int64 new_add = rand48_seed;
114
115 // calculate a = a^numStreams
116 rand48_seed = rand48_mult;
117 rand48_add = 0;
118 for ( i = 1; i < numStreams; ++i ) skip();
119 rand48_mult = rand48_seed;
120
121 rand48_add = new_add;
122 rand48_seed = save_seed;
123
124 second_gaussian = 0;
125 second_gaussian_waiting = false;
126 print("END SPLIT\n");
127 }

129 // return a number uniformly distributed between 0 and 1

131 skip();

136 long poisson(BigReal lambda) {
137 const BigReal l = exp(-lambda);
138 long k = 0;
139 BigReal p = uniform();
140
141 while (p >= l) {
142 p *= uniform();
143 k = k + 1;
144 }
145
146 return k;
147 }

149 // return a number from a standard gaussian distribution
150 HOST DEVICE inline BigReal gaussian(void) {
151 BigReal fac, r, v1, v2;
152
153 if (second_gaussian_waiting) {
154 second_gaussian_waiting = false;
155 return second_gaussian;
156 } else {
157 r = 2.; // r >= 1.523e-8 ensures abs result < 6
158 while (r >=1. || r < 1.523e-8) { // make sure we are within unit circle
159 v1 = 2.0 * uniform() - 1.0;
160 v2 = 2.0 * uniform() - 1.0;
161 r = v1*v1 + v2*v2;
162 // printf("r %f\n", r);
163 }
164 fac = sqrt(-2.0 * log(r)/r);
165 // now make the Box-Muller transformation to get two normally
166 // distributed random numbers. Save one and return the other.
167 second_gaussian_waiting = true;
168 second_gaussian = v1 * fac;
169 return v2 * fac;
170 }
171
172 }

185 // randomly order an array of whatever
186 template <class Elem> void reorder(Elem *a, int n) {
187 for ( int i = 0; i < (n-1); ++i ) {
188 int ie = i + ( integer() % (n-i) );
189 if ( ie == i ) continue;
190 const Elem e = a[ie];
191 a[ie] = a[i];
192 a[i] = e;
193 }
194 }
