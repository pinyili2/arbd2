92 // split into numStreams different steams and take stream iStream
93 void split(int iStream, int numStreams) {
94 int i;
95
96 // make sure that numStreams is odd to ensure maximum period
97 numStreams |= 1;
98
99 // iterate to get to the correct stream
100 for ( i = 0; i < iStream; ++i ) skip();
101
102 // save seed and add so we can use skip() for our calculations
103 int64 save_seed = rand48_seed;
104
105 // calculate c *= ( 1 + a + ... + a^(numStreams-1) )
106 rand48_seed = rand48_add;
107 for ( i = 1; i < numStreams; ++i ) skip();
108 int64 new_add = rand48_seed;
109
110 // calculate a = a^numStreams
111 rand48_seed = rand48_mult;
112 rand48_add = 0;
113 for ( i = 1; i < numStreams; ++i ) skip();
114 rand48_mult = rand48_seed;
115
116 rand48_add = new_add;
117 rand48_seed = save_seed;
118
119 second_gaussian = 0;
120 second_gaussian_waiting = 0;
121 print("END SPLIT\n");
122 }

126 skip();

131 long poisson(BigReal lambda) {
132 const BigReal l = exp(-lambda);
133 long k = 0;
134 BigReal p = uniform();
135
136 while (p >= l) {
137 p *= uniform();
138 k = k + 1;
139 }
140
141 return k;
142 }

144 // return a number from a standard gaussian distribution
145 HOST DEVICE inline BigReal gaussian(void) {
146 BigReal fac, r, v1, v2;
147
148 if (second_gaussian_waiting) {
149 second_gaussian_waiting = 0;
150 return second_gaussian;
151 } else {
152 r = 2.; // r >= 1.523e-8 ensures abs result < 6
153 while (r >=1. || r < 1.523e-8) { // make sure we are within unit circle
154 v1 = 2.0 * uniform() - 1.0;
155 v2 = 2.0 * uniform() - 1.0;
156 r = v1*v1 + v2*v2;
157 // printf("r %f", r);
158 }
159 fac = sqrt(-2.0 * log(r)/r);
160 // now make the Box-Muller transformation to get two normally
161 // distributed random numbers. Save one and return the other.
162 second_gaussian_waiting = 1;
163 second_gaussian = v1 * fac;
164 return v2 * fac;
165 }
166
167 }

180 // randomly order an array of whatever
181 template <class Elem> void reorder(Elem *a, int n) {
182 for ( int i = 0; i < (n-1); ++i ) {
183 int ie = i + ( integer() % (n-i) );
184 if ( ie == i ) continue;
185 const Elem e = a[ie];
186 a[ie] = a[i];
187 a[i] = e;
188 }
