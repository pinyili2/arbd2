32 const Vector3 posc = d->ind3;
33 const Vector3 posd = d->ind4;

35 const Vector3 ab = sys->wrapDiff(posa - posb);
36 const Vector3 bc = sys->wrapDiff(posb - posc);
37 const Vector3 cd = sys->wrapDiff(posc - posd);

39 //const float distab = ab.length();

41 //const float distcd = cd.length();

43 Vector3 crossABC = ab.cross(bc);
44 Vector3 crossBCD = bc.cross(cd);
45 Vector3 crossX = bc.cross(crossABC);
46
47 const float cos_phi = crossABC.dot(crossBCD) / (crossABC.length() * crossBCD.length());
48 const float sin_phi = crossX.dot(crossBCD) / (crossX.length() * crossBCD.length());
49
50 const float angle = -atan2(sin_phi, cos_phi);
51
52 float energy = 0.0f;
53 float force = 0.0f;
54
55 Vector3 f1, f2, f3; // forces
56 f1 = -distbc * crossABC.rLength2() * crossABC;
57 f3 = -distbc * crossBCD.rLength2() * crossBCD;
58 f2 = -(ab.dot(bc) * bc.rLength2()) * f1 - (bc.dot(cd) * bc.rLength2()) * f3;
59
60 // Shift "angle" by "PI" since -PI < dihedral < PI
61 // And our tabulated potential data: 0 < angle < 2 PI
62 float t = (angle + BD_PI) * angle_step_inv;
63 int home = (int) floorf(t);
64 t = t - home;
65
66 home = home % size;
67 int home1 = (home + 1) % size;
68
69 //================================================
70 // Linear interpolation
71 float U0 = pot[home]; // Potential
72 float dU = pot[home1] - U0; // Change in potential
73
74 energy = dU * t + U0;
75 force = -dU * angle_step_inv;
76 //================================================
77 // TODO: add an option for cubic interpolation
78
79 if (crossABC.rLength() > 1.0f || crossBCD.rLength() > 1.0f)
80 // avoid singularity when one angle is straight
81 force = 0.0f;
82
83 f1 *= force;
84 f2 *= force;
85 f3 *= force;
86
87 switch (index) {
88 // Return energy and forces to appropriate particles
89 case 1: return EnergyForce(energy, f1); // a
90 case 2: return EnergyForce(energy, f2 - f1); // b
91 case 3: return EnergyForce(energy, f3 - f2); // c
92 case 4: return EnergyForce(energy, -f3); // d
93 default: return EnergyForce(0.0f, Vector3(0.0f));
94 }
