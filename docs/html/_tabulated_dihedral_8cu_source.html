10TabulatedDihedralPotential::TabulatedDihedralPotential(String fileName) : fileName(fileName), size(0) {
11 FILE* inp = fopen(fileName.val(), "r");
12 if (inp == NULL) {
13 printf("TabulatedDihedralPotential: could not open file '%s'\n", fileName.val());
14 exit(-1);
15 }
16 char line[256];
17 int capacity = 256;
18 float* angle = new float[capacity];
19 pot = new float[capacity];
20 while(fgets(line, 256, inp)) {
21 String s(line);
22 int numTokens = s.tokenCount();
23 String* tokenList = new String[numTokens];
24 s.tokenize(tokenList);
25
26 // Legitimate TABULATED Dihedral inputs have 2 tokens
27 // ANGLE | VALUE
28 // Any angle input line without exactly 2 tokens should be discarded
29 if (numTokens != 2) {
30 printf("Invalid Dihedral input line: %s\n", line);
31 continue;
32 }
33
34 // Discard any empty line
35 if (tokenList == NULL) {
36 printf("Empty Dihedral input line: %s\n", line);
37 continue;
38 }
39
40 if (size >= capacity) {
41 float* temp = angle;
42 float* temp2 = pot;
43 capacity *= 2;
44 angle = new float[capacity];
45 pot = new float[capacity];
46 for (int i = 0; i < size; i++) {
47 angle[i] = temp[i];
48 pot[i] = temp2[i];
49 }
50 delete[] temp;
51 delete[] temp2;
52 }
53 angle[size] = atof(tokenList[0].val());
54 pot[size++] = atof(tokenList[1].val());
55 }
56 // units "1/deg" "1/radian" *57.29578
57 float deltaAngle = (angle[size-1]-angle[0])/(size-1);
58 assert( deltaAngle > 0 );
59 assert( size*deltaAngle >= 360 );
60
61 float tmp[size];
62 for (int j = 0; j < size; ++j) {
63 // reorganize data so that the angle goes from [-Pi,Pi+delta)
64 float a = -180.0f - angle[0] + j*deltaAngle;
65 while (a < 0) a += 360.0f;
66 while (a >= size*deltaAngle) a -= 360.0f;
67 int i = round( a / deltaAngle );
68 assert(i >= 0);
69 assert(i < size);
70 tmp[j] = pot[i];
71 }
72 for (int i = 0; i < size; ++i) pot[i] = tmp[i];
73
74 angle_step_inv = 57.29578f / deltaAngle;
75
76 delete[] angle;
77 fclose(inp);
78}
