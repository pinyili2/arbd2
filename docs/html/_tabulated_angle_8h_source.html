26 HOST DEVICE inline EnergyForce computeOLD(Angle* a, Vector3* pos, BaseGrid* sys, int index) {
27 // First, we must find the actual angle we're working with.
28 // Grab the positions of each particle in the angle
29 const Vector3 posa = pos[a->ind1];
30 const Vector3 posb = pos[a->ind2];
31 const Vector3 posc = pos[a->ind3];
32
33 // The vectors between each pair of particles
34 const Vector3 ab = sys->wrapDiff(posa - posb);
35 const Vector3 bc = sys->wrapDiff(posb - posc);
36 const Vector3 ac = sys->wrapDiff(posc - posa);
37
38 // Find the distance between each pair of particles
39 const float distab = ab.length();
40 const float distbc = bc.length();
41 const float distac = ac.length();
42
43 // Find the cosine of the angle we want - <ABC
44 float cos = (distbc * distbc + distab * distab - distac * distac) / (2.0f * distbc * distab);
45
46 // If the cosine is illegitimate, set it to 1 or -1 so that acos won't fail
47 if (cos < -1.0f) cos = -1.0f;
48 if (cos > 1.0f) cos = 1.0f;
49
50 // Find the sine while we're at it.
51 float sin = sqrtf(1.0f - cos*cos);
52
53 // Now we can use the cosine to find the actual angle (in radians)
54 float angle = acos(cos);
55
56 // tableAngle is divided into units of angle_step length
57 // 'convertedAngle' is the angle, represented in these units
58 float convertedAngle = angle * angle_step_inv;
59
60 // tableAngle[0] stores the potential at angle_step
61 // tableAngle[1] stores the potential at angle_step * 2, etc.
62 // 'home' is the index after which 'convertedAngle' would appear if it were stored in the table
63
64 int home = int(floor(convertedAngle));
65
66 // diffHome is the distance between the convertedAngle and the home index
67 float diffHome = convertedAngle - home;
68
69 // Linear interpolation for the potential
70 float pot0 = pot[home];
71 float delta_pot = pot[(home+1) % size] - pot0;
72 float energy = (delta_pot * angle_step_inv) * diffHome + pot0;
73 float diff = -delta_pot * angle_step_inv;
74 diff /= sin;
75
76 // Don't know what these are for, so I didn't bother giving them better names.
77 // Sorry, future person.
78 float c1 = diff / distab;
79 float c2 = diff / distbc;
80
81 // Calculate the forces
82 Vector3 force1 = c1 * (ab * (cos / distab) - bc / distbc); // force on particle 1
83 Vector3 force3 = c2 * (bc * (cos / distbc) - ab / distab); // force on particle 3
84 Vector3 force2 = -(force1 + force3); // the force on particle 2 (the central particle)
85
86 EnergyForce ret;
87 if (index == 1)
88 ret = EnergyForce(energy, force1);
89 if (index == 2)
90 ret = EnergyForce(energy, force2);
91 if (index == 3)
92 ret = EnergyForce(energy, force3);
93 return ret;
94 }
