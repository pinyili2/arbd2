351 // loop over potential grids
352 for (int i = 0; i < t->numPotGrids; ++i) {
353 if (numParticles[i] <= 0) continue;
354 const int nb = (numParticles[i]/NUMTHREADS)+1;
355 int idx = t->potential_grid_idx[i];
356 Vector3 c = getOrientation()*t->RBC->grids[idx].getOrigin() + getPosition();
357
358 // Sum and apply forces and torques
359 //Vector3 f = Vector3(0.0f);
360 ForceEnergy f = ForceEnergy(0.f,0.f);
361 Vector3 torq = Vector3(0.0f);
362 for (int k = 0; k < nb; ++k) {
363 int j = forcestorques_offset[fto_idx]+2*k;
364 f = f + forcestorques[j];
365 torq = torq + forcestorques[j+1].f;
366 }

369 torq = -torq + (sys->wrapDiff(getPosition()-c)).cross( f.f );
370 addForce( -f.f );
371 addTorque( torq );
372 addEnergy( f.e );
373 }

389void RigidBody::addLangevin(Vector3 w1, Vector3 w2)
390{
391 Vector3 transForceCoeff = Vector3::element_sqrt( 2. * Temp * t->mass*t->transDamping / timestep );
392 Vector3 rotTorqueCoeff = Vector3::element_sqrt( 2. * Temp * Vector3::element_mult( t->inertia,t->rotDamping) / timestep );
393
394 Force f = Vector3::element_mult(transForceCoeff,w1) -
395 Vector3::element_mult(t->transDamping, orientation.transpose()*momentum) * 10000;
396
397 Force torq = Vector3::element_mult(rotTorqueCoeff,w2) -
398 Vector3::element_mult(t->rotDamping, angularMomentum) * 10000;
399
400 f = orientation * f;
401 torq = orientation * torq;
402
403 addForce(f);
404 addTorque(torq);
405}

492 position = sys->wrap( position );

495 Vector3 rotationOffset = Vector3::element_mult( (rotDiffusion / Temp) , orientation.transpose() * torque * timestep) +

499 orientation = orientation * (Rz(rotationOffset.z * 0.5) * Ry(rotationOffset.y * 0.5) * Rx(rotationOffset.x)

513void RigidBody::applyRotation(const Matrix3& R) {
514 angularMomentum = R * angularMomentum;
515 // According to DLM, but rotations work the wrong way; I think DLM update is wrong
516 // orientation = orientation * R.transpose();
517
518 // This makes sense: apply a rotation in the body frame followed by a transformation from body to lab frame
519 // Also works in statistical test
520 // Consistent with www.archer.ac.uk/documentation/white-papers/lammps-elba/lammps-ecse.pdf
521 orientation = orientation * R;
522 orientation.normalized();
523}

525// Rotations about axes
526// for very small angles 10^-8, cos^2+sin^2 != 1
527// concerned about the accumulation of errors in non-unitary transformations!
528Matrix3 RigidBody::Rx(BigReal t) {
529 BigReal qt = 0.25*t*t; // for approximate calculations of sin(t) and cos(t)
530 BigReal cos = (1-qt)/(1+qt);
531 BigReal sin = t/(1+qt);
532
533 return Matrix3(
534 1.0f, 0.0f, 0.0f,
535 0.0f, cos, -sin,
536 0.0f, sin, cos);
537}
538Matrix3 RigidBody::Ry(BigReal t) {
539 BigReal qt = 0.25*t*t; // for approximate calculations of sin(t) and cos(t)
540 BigReal cos = (1-qt)/(1+qt);
541 BigReal sin = t/(1+qt);
542
543 return Matrix3(
544 cos, 0.0f, sin,
545 0.0f, 1.0f, 0.0f,
546 -sin, 0.0f, cos);
547}
548Matrix3 RigidBody::Rz(BigReal t) {
549 BigReal qt = 0.25*t*t; // for approximate calculations of sin(t) and cos(t)
550 BigReal cos = (1-qt)/(1+qt);
551 BigReal sin = t/(1+qt);
552
553 return Matrix3(
554 cos, -sin, 0.0f,
555 sin, cos, 0.0f,
556 0.0f, 0.0f, 1.0f);
557}
558Matrix3 RigidBody::eulerToMatrix(const Vector3 e) {
559 // convert euler angle input to rotation matrix
